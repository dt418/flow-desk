### Task 1: Rate-limit middleware

**Files:**
- Create: `apps/api/src/shared/middleware/rate-limit.ts`

- [ ] **Step 1: Create `rate-limit.ts`**

```ts
import type { Context, MiddlewareHandler } from 'hono';
import { redis } from '../lib/redis';
import { env } from '../lib/env';
import { RateLimitError } from '../errors';
import { logger } from '../lib/logger';

export interface RateLimitOptions {
  windowSec: number;
  max: number;
  scope: string;
  keyBy: 'ip' | 'user';
}

function clientIp(c: Context): string {
  return (
    c.req.header('x-forwarded-for')?.split(',')[0]?.trim() ||
    c.req.header('x-real-ip') ||
    'unknown'
  );
}

function actorKey(c: Context, keyBy: 'ip' | 'user'): string {
  if (keyBy === 'user') {
    const auth = c.get('auth');
    return auth?.user?.id ? `u:${auth.user.id}` : `ip:${clientIp(c)}`;
  }
  return `ip:${clientIp(c)}`;
}

export function rateLimit(opts: RateLimitOptions): MiddlewareHandler {
  return async (c, next) => {
    if (env.NODE_ENV === 'test' || env.SKIP_RATE_LIMIT === '1') {
      await next();
      return;
    }
    const now = Math.floor(Date.now() / 1000);
    const windowStart = Math.floor(now / opts.windowSec) * opts.windowSec;
    const resetEpoch = windowStart + opts.windowSec;
    const key = `rl:${opts.scope}:${actorKey(c, opts.keyBy)}:${windowStart}`;
    const count = await redis.incr(key);
    if (count === 1) {
      await redis.expire(key, opts.windowSec);
    }
    const remaining = Math.max(0, opts.max - count);
    c.header('X-RateLimit-Limit', String(opts.max));
    c.header('X-RateLimit-Remaining', String(remaining));
    c.header('X-RateLimit-Reset', String(resetEpoch));
    if (count > opts.max) {
      const retryAfter = resetEpoch - now;
      logger.warn({ scope: opts.scope, count, max: opts.max }, 'rate limit exceeded');
      throw new RateLimitError(`Rate limit exceeded for ${opts.scope}`, { retryAfterSec: retryAfter });
    }
    await next();
  };
}
```

### Task 4: LLM provider — timeout + retry

**Files:**

- [ ] **Step 1: Read current `llm-provider.ts`**

Find the `fetch` call and the `chatJSON` wrapper.

- [ ] **Step 2: Replace the fetch wrapper with timeout+retry**

Inside the file, wrap the existing `fetch(...)` block (the body that builds request + calls fetch + parses) in a `withTimeoutRetry` helper. Use this pattern:

```ts
async function callOnce(body: unknown, signal: AbortSignal): Promise<Response> {
  const res = await fetch(`${env.LLM_BASE_URL}/chat/completions`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${env.LLM_API_KEY}`,
    },
    body: JSON.stringify(body),
    signal,
  });
  return res;
}

async function withTimeoutRetry<T>(fn: (signal: AbortSignal) => Promise<T>): Promise<T> {
  const TIMEOUT_MS = 30_000;
  const attempts = 2;
  for (let i = 1; i <= attempts; i++) {
    const ctrl = new AbortController();
    const timer = setTimeout(() => ctrl.abort(), TIMEOUT_MS);
    try {
      return await fn(ctrl.signal);
    } catch (err: any) {
      const retriable =
        err?.name === 'AbortError' ||
        (err instanceof LLMError && (err.status === 502 || err.status === 503 || err.status === 500));
      if (i < attempts && retriable) {
        logger.warn({ attempt: i, err: err?.message }, 'llm call retried');
        await new Promise((r) => setTimeout(r, 500));
        continue;
      }
      throw err;
    } finally {
      clearTimeout(timer);
    }
  }
  throw new Error('unreachable');
}
```

Then replace the inner non-OK throw so it produces a status-aware `LLMError` (e.g. `new LLMError(\`Upstream ${res.status}: ${bodyText.slice(0, 200)}\`)`).

```ts
import type { Server } from 'socket.io';
import { logger } from './logger';

let _io: Server | null = null;

export function setIo(io: Server) {
  _io = io;
  logger.info('socket-events: io singleton set');
}

export function getIo(): Server | null {
  return _io;
}

function emit(ns: string, room: string, event: string, payload: unknown) {
  if (!_io) {
    logger.debug({ ns, room, event }, 'socket-events: emit skipped (no io)');
    return;
  }
  _io.of(ns).to(room).emit(event, payload);
}

export function emitTaskCreated(workspaceId: string, task: unknown) {
  emit('tasks', `workspace:${workspaceId}`, 'task:created', task);
}

export function emitTaskUpdated(workspaceId: string, task: unknown) {
  emit('tasks', `workspace:${workspaceId}`, 'task:updated', task);
}

export function emitTaskDeleted(workspaceId: string, taskId: string) {
  emit('tasks', `workspace:${workspaceId}`, 'task:deleted', { id: taskId });
}

export function emitTaskMoved(workspaceId: string, task: unknown) {
  emit('tasks', `workspace:${workspaceId}`, 'task:moved', task);
}

export function emitCommentCreated(workspaceId: string, comment: unknown) {
  emit('collab', `workspace:${workspaceId}`, 'comment:created', comment);
}

export function emitNotification(userId: string, notification: unknown) {
  emit('notifications', `user:${userId}`, 'notification:new', notification);
}
```
